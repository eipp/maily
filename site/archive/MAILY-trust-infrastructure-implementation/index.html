    Overview¶

    The MAILY Trust Infrastructure introduces blockchain technology and advanced AI to create a transparent, verifiable system for email marketing, establishing trust between marketers and recipients while driving organic growth through engagement incentives.

    Architecture Overview¶

    The architecture introduces:

    1. Blockchain Layer: Smart contracts for token minting, trust certificates, and rewards
    2. AI Trust Layer: Enhanced AI models for personalization and trend prediction
    3. User Engagement Layer: Referral systems, trust scores, and staking mechanics

    // SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/AccessControl.sol";
import "@chainlink/contracts/src/v0.8/interfaces/AggregatorV3Interface.sol";
import "@layerzerolabs/solidity-examples/contracts/token/oft/v2/OFTV2.sol";

contract MailyToken is OFTV2, AccessControl {
    bytes32 public constant MINTER_ROLE = keccak256("MINTER_ROLE");
    bytes32 public constant ORACLE_ROLE = keccak256("ORACLE_ROLE");

    uint256 public constant MAX_SUPPLY = 1_000_000_000 * 10**18; // 1 billion tokens
    mapping(uint256 => bool) public campaignProcessed;
    mapping(address => mapping(uint256 => uint256)) public performanceRewards;

    AggregatorV3Interface public emailMetricsOracle;

    event CampaignRewardIssued(address indexed marketer, uint256 indexed campaignId, uint256 amount);

    constructor(
        string memory _name,
        string memory _symbol,
        address _layerZeroEndpoint,
        address _oracleAddress
    ) OFTV2(_name, _symbol, 18, _layerZeroEndpoint) {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(MINTER_ROLE, msg.sender);
        emailMetricsOracle = AggregatorV3Interface(_oracleAddress);

        // Mint initial supply
        _mint(msg.sender, MAX_SUPPLY / 2);
    }

    function mintBasedOnPerformance(
        address marketer,
        uint256 campaignId,
        uint256 openRate,
        uint256 clickRate
    ) external onlyRole(MINTER_ROLE) {
        require(!campaignProcessed[campaignId], "Campaign already processed");

        // Calculate reward based on open rate and click rate
        uint256 reward = (openRate * 5 + clickRate * 10) * 10**16;
        require(totalSupply() + reward <= MAX_SUPPLY, "Exceeds max supply");

        _mint(marketer, reward);
        performanceRewards[marketer][campaignId] = reward;
        campaignProcessed[campaignId] = true;

        emit CampaignRewardIssued(marketer, campaignId, reward);
    }

    function updateOracleAddress(address _oracleAddress) external onlyRole(DEFAULT_ADMIN_ROLE) {
        emailMetricsOracle = AggregatorV3Interface(_oracleAddress);
    }

    // Override required function in AccessControl
    function supportsInterface(bytes4 interfaceId) public view virtual override(AccessControl, OFTV2) returns (bool) {
        return super.supportsInterface(interfaceId);
    }
}

    Create a custom Chainlink External Adapter to fetch and verify email metrics:

    // apps/blockchain/oracles/email-metrics-adapter.js
const { Requester, Validator } = require('@chainlink/external-adapter');

// Define custom error messages
const customError = (data) => {
  if (data.Response === 'Error') return true;
  return false;
};

// Define custom parameters
const customParams = {
  campaignId: ['campaignId', 'campaign'],
  endpoint: false
};

// Create request handler
const createRequest = (input, callback) => {
  // Validate input data
  const validator = new Validator(input, customParams);
  const jobRunID = validator.validated.id;
  const campaignId = validator.validated.data.campaignId;
  const endpoint = validator.validated.data.endpoint || 'metrics';
  const url = `https://api.maily.app/campaigns/${campaignId}/${endpoint}`;

  // Set request options
  const params = {
    campaignId
  };

  // Make the request to the API
  Requester.request(url, params)
    .then(response => {
      response.data.result = {
        openRate: response.data.metrics.openRate * 100, // Convert to basis points (0.15 -> 15)
        clickRate: response.data.metrics.clickRate * 100, // Convert to basis points
        bounceRate: response.data.metrics.bounceRate * 100, // Convert to basis points
        timestamp: Math.floor(Date.now() / 1000)
      };

      callback(response.status, Requester.success(jobRunID, response));
    })
    .catch(error => {
      callback(500, Requester.errored(jobRunID, error));
    });
};

// Export handler for serverless function
module.exports.createRequest = createRequest;

import os
import json
from typing import Dict, Any
import httpx
from eth_account import Account
from eth_account.signers.local import LocalAccount
from web3 import Web3

from ..models import Campaign, CampaignMetrics

class BlockchainService:
    def __init__(self):
        self.private_key = os.getenv("BLOCKCHAIN_PRIVATE_KEY")
        self.account: LocalAccount = Account.from_key(self.private_key)
        self.w3 = Web3(Web3.HTTPProvider(os.getenv("ETHEREUM_RPC_URL")))

        # Load ABI from file
        with open("contracts/MailyToken.json") as f:
            contract_json = json.load(f)
            self.abi = contract_json["abi"]

        self.contract_address = os.getenv("MAILY_TOKEN_ADDRESS")
        self.contract = self.w3.eth.contract(address=self.contract_address, abi=self.abi)

    async def record_campaign_metrics(self, campaign_id: int, metrics: CampaignMetrics, marketer_address: str):
        """Record campaign metrics to blockchain via Chainlink Oracle"""

        # Call Oracle to update metrics (this triggers the Oracle to fetch and verify data)
        oracle_url = os.getenv("CHAINLINK_ORACLE_URL")
        async with httpx.AsyncClient() as client:
            response = await client.post(oracle_url, json={
                "campaignId": campaign_id,
                "marketerAddress": marketer_address,
                "openRate": metrics.open_rate,
                "clickRate": metrics.click_rate
            })

            if response.status_code != 200:
                raise Exception(f"Failed to record metrics on-chain: {response.text}")

            return response.json()

class CampaignService:

        self.blockchain_service = BlockchainService()

logger = logging.getLogger(__name__)

class AITrustAgent:

        """Initialize the AI Trust Agent.

    def _load_personalization_model(self):
        """Load the TensorFlow personalization model."""
        try:
            model_path = os.path.join(os.path.dirname(__file__), "models/personalization_model")
            if os.path.exists(model_path):
                self.personalization_model = tf.keras.models.load_model(model_path)
                logger.info("Personalization model loaded successfully")
            else:
                logger.warning("Personalization model not found, creating new model")
                self._create_personalization_model()
        except Exception as e:
            logger.error(f"Failed to load personalization model: {str(e)}")
            self._create_personalization_model()

    def _create_personalization_model(self):
        """Create a new TensorFlow personalization model."""
        # Create a simple neural network for personalization
        model = tf.keras.Sequential([
            tf.keras.layers.Input(shape=(10,)),  # 10 features per user
            tf.keras.layers.Dense(64, activation='relu'),
            tf.keras.layers.Dropout(0.2),
            tf.keras.layers.Dense(32, activation='relu'),
            tf.keras.layers.Dropout(0.2),
            tf.keras.layers.Dense(16, activation='relu'),
            tf.keras.layers.Dense(4, activation='softmax')  # 4 personalization categories
        ])

        model.compile(
            optimizer='adam',
            loss='sparse_categorical_crossentropy',
            metrics=['accuracy']
        )

        logger.info("Created new personalization model")

            "personalization_category": category,
            "confidence": confidence,
            "features": features.tolist(),
            "recommended_approach": self._get_personalization_approach(category)
        }

    async def generate_personalized_content(
        self,
        campaign_id: int,
        user_profile: Dict[str, Any],
        campaign_data: Dict[str, Any]
    ) -> Dict[str, Any]:
        """Generate personalized content for a user.

        Args:
            campaign_id: Campaign ID
            user_profile: User personalization profile
            campaign_data: Campaign data

        Returns:
            Dictionary containing personalized content
        """
        try:
            # Create prompt template
            template = """
            Create a personalized email for a recipient with the following profile:
            - Personalization category: {category}
            - Content style preference: {content_style}
            - Subject style preference: {subject_style}
            - CTA style preference: {cta_style}

            Campaign details:
            - Campaign objective: {objective}
            - Main message: {message}
            - Call to action: {cta}
            - Target audience: {audience}

            Generate:
            1. A personalized subject line
            2. Personalized email body in HTML format
            3. A short plain text version

            Format your response as JSON with keys: "subject", "html_body", "plain_text"
            """

            prompt = PromptTemplate(
                input_variables=["category", "content_style", "subject_style", "cta_style",
                                "objective", "message", "cta", "audience"],
                template=template
            )

            # Create LLM chain
            chain = LLMChain(llm=self.llm, prompt=prompt)

            # Run chain
            approach = user_profile["recommended_approach"]
            result = await chain.arun(
                category=user_profile["personalization_category"],
                content_style=approach["content_style"],
                subject_style=approach["subject_style"],
                cta_style=approach["cta_style"],
                objective=campaign_data.get("objective", ""),
                message=campaign_data.get("message", ""),
                cta=campaign_data.get("cta", ""),
                audience=campaign_data.get("audience", "")
            )

            # Parse JSON result
            content = json.loads(result)

            return {
                "campaign_id": campaign_id,
                "user_profile": user_profile,
                "personalized_content": content,
                "trust_score": self._calculate_trust_score(user_profile, campaign_data)
            }

        except Exception as e:
            logger.error(f"Failed to generate personalized content: {str(e)}")
            raise

    def _calculate_trust_score(self, user_profile: Dict[str, Any], campaign_data: Dict[str, Any]) -> int:
        """Calculate trust score for a personalized campaign.

        Args:
            user_profile: User personalization profile
            campaign_data: Campaign data

        Returns:
            Trust score (0-100)
        """
        score = 0

        # Base score based on personalization confidence
        score += int(user_profile.get("confidence", 0) * 50)

        # Additional score based on content alignment
        category = user_profile.get("personalization_category")
        if category == "informational" and "educational" in campaign_data.get("tags", []):
            score += 15
        elif category == "promotional" and "offer" in campaign_data.get("tags", []):
            score += 15
        elif category == "relationship" and "newsletter" in campaign_data.get("tags", []):
            score += 15
        elif category == "transactional" and "receipt" in campaign_data.get("tags", []):
            score += 15

        # Adjust based on unsubscribe risk
        unsubscribe_risk = user_profile.get("unsubscribe_risk", 0)
        score -= int(unsubscribe_risk * 30)

        # Ensure score is in range 0-100
        return max(0, min(100, score))

class CampaignService:
    def __init__(self, settings: Settings):
        self.settings = settings
        self.blockchain_service = BlockchainService()
        self.ai_trust_agent = AITrustAgent(settings)

    async def create_campaign_with_personalization(
        self,
        campaign_data: Dict[str, Any],
        user_id: int
    ) -> Dict[str, Any]:
        """Create a campaign with personalized content for recipients.

        Args:
            campaign_data: Campaign data
            user_id: User ID

        Returns:
            Dictionary containing campaign creation result
        """
        # Create base campaign
        campaign_id = await self.create_campaign(campaign_data)

        # Get recipient list
        recipients = await self.get_recipients(campaign_data)

        # Generate personalized content for each recipient
        personalized_content = {}
        trust_scores = {}

        for recipient in recipients:
            # Analyze user for personalization
            user_profile = await self.ai_trust_agent.analyze_user(recipient["id"])

            # Generate personalized content
            content = await self.ai_trust_agent.generate_personalized_content(
                campaign_id=campaign_id,
                user_profile=user_profile,
                campaign_data=campaign_data
            )

            personalized_content[recipient["id"]] = content["personalized_content"]
            trust_scores[recipient["id"]] = content["trust_score"]

        # Store personalized content and trust scores
        await self.store_personalized_content(campaign_id, personalized_content, trust_scores)

        return {
            "campaign_id": campaign_id,
            "recipient_count": len(recipients),
            "personalization_success": True,
            "avg_trust_score": sum(trust_scores.values()) / max(len(trust_scores), 1)
        }

    Smart Contract for Trust Certificates¶

    // SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

import "@openzeppelin/contracts/access/AccessControl.sol";

contract TrustCertificate is AccessControl {
    bytes32 public constant ISSUER_ROLE = keccak256("ISSUER_ROLE");

    struct Certificate {
        address marketer;
        uint256 campaignId;
        bytes32 metricsHash;
        uint256 timestamp;
        bool revoked;
    }

    mapping(bytes32 => Certificate) public certificates;
    mapping(address => mapping(uint256 => bytes32)) public marketerCampaignCerts;

    event CertificateIssued(
        bytes32 indexed certId,
        address indexed marketer,
        uint256 indexed campaignId,
        bytes32 metricsHash
    );
    event CertificateRevoked(bytes32 indexed certId);

    constructor() {
        _grantRole(DEFAULT_ADMIN_ROLE, msg.sender);
        _grantRole(ISSUER_ROLE, msg.sender);
    }

    function issueCertificate(
        address marketer,
        uint256 campaignId,
        bytes32 metricsHash
    ) external onlyRole(ISSUER_ROLE) returns (bytes32) {
        bytes32 certId = keccak256(abi.encodePacked(marketer, campaignId, metricsHash, block.timestamp));

        certificates[certId] = Certificate({
            marketer: marketer,
            campaignId: campaignId,
            metricsHash: metricsHash,
            timestamp: block.timestamp,
            revoked: false
        });

        marketerCampaignCerts[marketer][campaignId] = certId;

        emit CertificateIssued(certId, marketer, campaignId, metricsHash);

        return certId;
    }

    function revokeCertificate(bytes32 certId) external onlyRole(ISSUER_ROLE) {
        require(certificates[certId].timestamp > 0, "Certificate does not exist");
        require(!certificates[certId].revoked, "Certificate already revoked");

        certificates[certId].revoked = true;

        emit CertificateRevoked(certId);
    }

    function verifyCertificate(bytes32 certId) external view returns (
        address marketer,
        uint256 campaignId,
        bytes32 metricsHash,
        uint256 timestamp,
        bool revoked
    ) {
        Certificate memory cert = certificates[certId];
        require(cert.timestamp > 0, "Certificate does not exist");

        return (
            cert.marketer,
            cert.campaignId,
            cert.metricsHash,
            cert.timestamp,
            cert.revoked
        );
    }

    function getCertificateForCampaign(
        address marketer,
        uint256 campaignId
    ) external view returns (bytes32) {
        return marketerCampaignCerts[marketer][campaignId];
    }
}

    Create a Certificate Service for the backend:

    # apps/api/services/certificate_service.py
import hashlib
import json
from typing import Dict, Any
from web3 import Web3
from eth_account import Account
from eth_account.signers.local import LocalAccount

from ..config import Settings
from ..models import TrustCertificate, Campaign, User

class CertificateService:
    def __init__(self, settings: Settings):
        self.settings = settings
        self.private_key = settings.BLOCKCHAIN_PRIVATE_KEY
        self.account: LocalAccount = Account.from_key(self.private_key)
        self.w3 = Web3(Web3.HTTPProvider(settings.ETHEREUM_RPC_URL))

        # Load ABI
        with open("contracts/TrustCertificate.json") as f:
            contract_json = json.load(f)
            self.abi = contract_json["abi"]

        self.contract_address = settings.TRUST_CERTIFICATE_ADDRESS
        self.contract = self.w3.eth.contract(address=self.contract_address, abi=self.abi)

    async def issue_certificate(
        self,
        campaign_id: int,
        marketer_address: str,
        metrics: Dict[str, Any]
    ) -> Dict[str, Any]:
        """Issue a trust certificate for a campaign.

        Args:
            campaign_id: Campaign ID
            marketer_address: Marketer's wallet address
            metrics: Campaign metrics

        Returns:
            Dictionary containing certificate information
        """
        # Generate metrics hash
        metrics_str = json.dumps(metrics, sort_keys=True)
        metrics_hash = hashlib.sha256(metrics_str.encode()).hexdigest()
        metrics_hash_bytes = bytes.fromhex(metrics_hash)

        # Prepare transaction
        nonce = self.w3.eth.get_transaction_count(self.account.address)

        # Build transaction
        tx = self.contract.functions.issueCertificate(
            marketer_address,
            campaign_id,
            metrics_hash_bytes
        ).build_transaction({
            'chainId': int(self.settings.ETHEREUM_CHAIN_ID),
            'gas': 200000,
            'gasPrice': self.w3.eth.gas_price,
            'nonce': nonce,
        })

        # Sign transaction
        signed_tx = self.account.sign_transaction(tx)

        # Send transaction
        tx_hash = self.w3.eth.send_raw_transaction(signed_tx.rawTransaction)

        # Wait for transaction receipt
        tx_receipt = self.w3.eth.wait_for_transaction_receipt(tx_hash)

        # Get certificate ID from event
        log = self.contract.events.CertificateIssued().process_receipt(tx_receipt)[0]
        cert_id = log['args']['certId'].hex()

        # Store certificate in database
        db_certificate = TrustCertificate(
            certificate_id=cert_id,
            user_id=campaign_id,  # This should be user ID in a real implementation
            campaign_id=campaign_id,
            metrics_hash=metrics_hash,
            blockchain_tx=tx_hash.hex()
        )

        # In a real implementation, save this to the database
        # await db.add(db_certificate)
        # await db.commit()

        return {
            "certificate_id": cert_id,
            "campaign_id": campaign_id,
            "marketer_address": marketer_address,
            "metrics_hash": metrics_hash,
            "transaction_hash": tx_hash.hex(),
            "verification_url": f"{self.settings.BLOCKCHAIN_EXPLORER_URL}/tx/{tx_hash.hex()}"
        }

    async def verify_certificate(self, cert_id: str) -> Dict[str, Any]:
        """Verify a trust certificate.

        Args:
            cert_id: Certificate ID

        Returns:
            Dictionary containing verification result
        """
        # Convert hex string to bytes32
        cert_id_bytes = bytes.fromhex(cert_id.replace("0x", ""))

        # Call contract
        result = self.contract.functions.verifyCertificate(cert_id_bytes).call()

        # Format result
        marketer, campaign_id, metrics_hash, timestamp, revoked = result

        return {
            "certificate_id": cert_id,
            "marketer_address": marketer,
            "campaign_id": campaign_id,
            "metrics_hash": metrics_hash.hex(),
            "timestamp": timestamp,
            "revoked": revoked,
            "verification_url": f"{self.settings.BLOCKCHAIN_EXPLORER_URL}/tx/{cert_id}"
        }
